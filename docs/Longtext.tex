\section{9/17/19 -- Relative Efficiency: Run-In vs Standard}

Compare precision of run-in design with `$R$'
pre-randomization observations to standard design.

To determine relative efficiency we calculate the
variance with one baseline and one followup and compare
to variance w/ $R$ baselines and one followup.

\subsection*{Sample size formula}

\marginmark{%
  \textbf{Model:}
  $x_{ij} = \alpha_i + \beta_j + e_{ij}$,
  \quad $i = 1,\ldots,m$, \; $j = 1,\ldots,K$ \\[4pt]
  $y_{ij} = \alpha_y + \beta_j + e_{ij}$ \\[4pt]
  $z_{ij} = x_{ij} - x_{i0} = y_{ij} - y_{i0}$}

\begin{equation}
  z_{\alpha/2}\,\sqrt{V(\bar{z}_1 - \bar{z}_2)}
  = z_{1-\beta}\,\sqrt{V(\bar{z}_1 - \bar{z}_2)} + \Delta
\end{equation}
or
\begin{equation}
  V(\bar{z}_1 - \bar{z}_2)
  = \frac{\Delta^2}
    {\bigl(z_{\alpha/2} - z_{1-\beta}\bigr)^2}
\end{equation}

Assuming equal samples per group size $m$:
\begin{equation}
  V(\bar{z}_1 - \bar{z}_2)
  = V(\bar{z}_1) + V(\bar{z}_2)
  = \frac{V(X_K - X_0)}{m}
    + \frac{V(Y_K - Y_0)}{m}
\end{equation}
or
\begin{equation}
  \frac{2\,V(X_K - X_0)}{m}
  = \frac{\Delta^2}
    {\bigl(z_{\alpha/2} - z_{1-\beta}\bigr)^2}
\end{equation}
\begin{equation}
  m = \frac{2\,V(X_{iK} - X_{i0})\,
    \bigl(z_{\alpha/2} - z_{1-\beta}\bigr)^2}
    {\Delta^2}
\end{equation}

\subsection*{One baseline \& one followup}

\begin{align}
  V(X_K - X_0)
  &= 2\sigma^2 - 2\rho\sigma^2
   = \sigma^2 + \sigma^2
     - 2\Cov(X_K,\, X_0) \notag \\
  &= 2\sigma^2(1 - \rho)
\end{align}

\subsection*{$R$ baselines, one followup}

\begin{align}
  V(X_K - \bar{X}_0)
  &= V(X_K) + V(\bar{X}_0)
     - 2\Cov(X_K,\, \bar{X}_0) \label{eq:Rbaseline} \\
  &= V(X_K) + V(\bar{X}_0)
     - 2\rho\,\sqrt{V(X_K)\,V(\bar{X}_0)} \notag
\end{align}
\begin{equation}
  V(X_K) = \sigma^2
\end{equation}
\begin{align}
  V(\bar{X}_0)
  &= V\!\left(\frac{X_{(K-1)} + \cdots + X_0}{K}\right)
   = \left(K\sigma^2
     + 2\rho\,\frac{K(K-1)}{2}\,\sigma^2\right)
     \frac{1}{K^2} \notag \\
  &= \bigl(K\sigma^2 + \rho\,K(K-1)\,\sigma^2\bigr)
     \frac{1}{K^2} \notag \\
  &= \frac{\sigma^2\bigl(1 + \rho(K-1)\bigr)}{K}
\end{align}

\section{9/17/19 (cont.) -- Efficiency Formula}

\begin{align}
  V(X_K - \bar{X}_0)
  &= \sigma^2
    + \frac{\sigma^2}{K}\bigl(1 + \rho(K-1)\bigr)
    - 2\rho\,\sqrt{\sigma^2 \cdot
      \frac{\sigma^2}{K}\bigl(1 + \rho(K-1)\bigr)}
    \notag \\[6pt]
  &= \sigma^2\!\left[
      1 + \frac{1}{K}\bigl(1 + \rho(K-1)\bigr)
      - 2\rho\,\sqrt{\frac{1+\rho(K-1)}{K}}
    \right] \notag \\[6pt]
  &= \sigma^2\!\left[
      \frac{K+1}{K}
      + \frac{K-1}{K}\,\rho
      - 2\rho\,\sqrt{\frac{1+\rho(K-1)}{K}}
    \right]
\end{align}

So the ratio is independent of $\sigma^2$.

\begin{equation}
  \boxed{
  \text{Inv.\ Efficiency}
  = \frac{2(1-\rho)}
    {\dfrac{K+1}{K}
     + \dfrac{K-1}{K}\,\rho
     - 2\rho\,\sqrt{\dfrac{1+\rho(K-1)}{K}}}
  }
\end{equation}

See report A11. Assumes constant correlation.

\section{9/21 -- OLS Centering Shortcut}

Consider the simplest case: OLS linear regression.
Looking for a centering shortcut.

\subsection*{Model}
\begin{equation}
  Y_i = \beta_0 + \beta_1 X_i + e_i
\end{equation}

OLS estimates:
$\hat{\bm{\beta}} = (X'X)^{-1}X'Y$
with $X = (\mathbf{1},\, \mathbf{x})$.
\begin{align}
  \hat{\beta}_0 &= \bar{Y} - \hat{\beta}_1 \bar{X}
  &
  \hat{\beta}_1 &= \frac{\sum(X_i - \bar{X})\,Y_i}
    {\sum(X_i - \bar{X})^2}
\end{align}
\[
  \Var(\hat{\bm{\beta}}) = (X'X)^{-1}
\]

\subsection*{Centering transformation}

Now transform
\[
  Z_i = \frac{X_i - \bar{X}}{s_x}, \qquad
  s_x = \sqrt{\frac{1}{n-1}\sum(X_i - \bar{X})^2}
\]

Observe:
\[
  \sum Z_i = 0, \qquad
  \Var(Z_i) = 1, \qquad
  \sum Z_i^2 = n - 1
\]

We can now work with $Z_i$ to obtain closed form
expressions for
$\hat{\bm{\beta}}_z = (\hat{\beta}_{1z},\,
\hat{\beta}_{2z})$ and $V(\hat{\bm{\beta}}_z)$.

\subsection*{Transformed model}
\begin{equation}
  Y_i = \beta_{0z} + \beta_{1z}\,Z_i + e_i
\end{equation}
\begin{equation}
  V(\hat{\bm{\beta}}_z) = (Z'Z)^{-1}
  = \begin{pmatrix}
      \frac{1}{n} & 0 \\
      0 & \frac{1}{n-1}
    \end{pmatrix}
\end{equation}
\begin{equation}
  \hat{\bm{\beta}}_z = (Z'Z)^{-1}Z'Y
  = \begin{pmatrix}
      \bar{Y} \\[4pt]
      \dfrac{\sum Z_i Y_i}{n-1}
    \end{pmatrix}
\end{equation}

\subsection*{Back-transformation to $X$ scale}

Thus given $\hat{\beta}_{1z}$ we can transform back
to $X$ with
\begin{equation}
  \hat{\beta}_{1x}
  = \hat{\beta}_{1z}
    \left[\frac{(n-1)\,\hat{s}_x}{\sum X_i^2}\right]
  = \hat{\beta}_{1z}
    \left[\frac{1}{s_x}\right]
\end{equation}
\begin{equation}
  \Var(\hat{\beta}_{1x})
  = \Var(\hat{\beta}_{1z})
    \left[\frac{(n-1)\,\hat{s}_x}{\sum X_i^2}\right]^2
  = \Var(\hat{\beta}_{1z})
    \left[\frac{1}{s_x}\right]^2
\end{equation}

\section{9/25/19 -- Mixed Effects Model}

\begin{equation}
  y_{ij} = \beta_0 + b_{0i}
    + (\beta_1 + b_{1i} + \beta_x g_i)\,t_j + e_{ij}
\end{equation}

Differencing from baseline:
\begin{align}
  z_{ij}
  &= y_{ij} - y_{i0}
   = (\beta_1 + b_{1i} + \beta_x g_i)(t_j - t_0)
     + e_{ij} - e_{i0} \notag \\
  &= (\beta_1 + b_{1i} + \beta_x g_i)\,d_j + \varepsilon_{ij}
\end{align}
where $d_j = t_j - t_0$ and
$\varepsilon_{ij} = e_{ij} - e_{i0}$.

In matrix form:
\begin{equation}
  \mathbf{z}_i = X_i \bm{\beta} + V_i b_i + \bm{\varepsilon}_i
\end{equation}
\[
  b_i \sim N(0,\, D), \qquad
  \bm{\varepsilon}_i \sim N(0,\, R)
\]

Let $w_i = \frac{1}{s_d}(d_i - \bar{d})$, \quad
$s_d^2 = \frac{1}{n-1}\sum(d_i - \bar{d})^2$.

\subsection*{GLS estimation}

From GLS (Thm?):
\begin{equation}
  \hat{\bm{\beta}}_{s_d}
  = (X^{*\prime} V^{*-1} X^*)^{-1}\,
    X^{*\prime} V^{*-1} X^*
\end{equation}
where $X^*$ is $X$ with $w$ replacing $d$.

\begin{alignat*}{2}
  \bm{\beta} &= (\beta_0,\, \beta_x) &\qquad
  b_i &= (b_{1i}) \\
  X_i &= (\mathbf{d},\, \mathbf{d})\;\text{[trt]}
   \;=\; (\mathbf{d},\, \mathbf{0})\;\text{[plc]}
  &\qquad
  D &= \sigma_b^2 \\
  && R &= \sigma^2 I
\end{alignat*}
\[
  \Var(\hat{\bm{\beta}})\,s_d^2
  = (X^{*\prime} V^{*-1} X^*)^{-1}
\]

\subsection*{Solution steps}

\begin{enumerate}
  \item Calculate $V_i^* = R_i + U_i' D\, U_i'$
  \item $V_i^{*-1}$
  \item $X_i' V_i^{*-1} X_i$ for $g_i = 1$
    and $g_i = 0$
  \item $F = \displaystyle\sum_{\trt}
    X_i' V_i^{*-1} X_i
    + \sum_{\plc} X_i' V_i^{*-1} X_i$
  \item $F^{-1}$
  \item $F^{-1}_{(2,2)} = V(\hat{\beta}_x^*)$, \quad
    $V(\hat{\beta}_x) = V(\hat{\beta}_x^*)\,
    \dfrac{1}{s_d^2}$
\end{enumerate}
